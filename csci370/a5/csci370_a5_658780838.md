    FROM Listings listing
    JOIN Properties property ON listing.folio = property.folio
    WHERE listing.askingPrice = (

    ```sql
    SELECT realtor.serviceName,
        realtor.expLevel,
        COUNT(closedListing.listingID) AS numSold
    FROM Realtors realtor
    LEFT JOIN ClosedListings closedListing ON realtor.RID = closedListing.soldBy
    GROUP BY realtor.RID, realtor.serviceName, realtor.expLevel;
    ```

5.

    ```sql
    SELECT property.civicAddress,
        property.builtYear,
        COUNT(*) AS numProjects
    FROM Properties property
    JOIN MajorProjects majorProperty ON property.folio = majorProperty.folio
    WHERE property.builtYear >= '2000'
    GROUP BY property.folio, property.civicAddress, property.builtYear
    HAVING COUNT(*) > 3
    ORDER BY property.builtYear DESC, numProjects DESC;
    ```
